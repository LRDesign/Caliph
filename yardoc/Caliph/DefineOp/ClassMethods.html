def define_chain_op(opname, klass)
  define_method(opname) do |other|
    unless CommandLine === other
      other = CommandLine.new(*[*other])
    end
    chain = nil
    if klass === self
      chain = self
    else
      chain = klass.new
      chain.add(self)
    end
    chain.add(other)
  end
end
